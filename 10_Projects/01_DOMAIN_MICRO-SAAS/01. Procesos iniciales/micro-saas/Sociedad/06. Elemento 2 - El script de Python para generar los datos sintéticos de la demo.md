# üß™ Elemento 2: Script de Python para Generar Datos Sint√©ticos (Demo)

Como a√∫n no tienes una base de datos real masiva, necesitas generar datos que "parezcan reales". Este script crea un archivo CSV con 1,000 registros donde hemos inyectado una **l√≥gica oculta**: los env√≠os en meses de verano, con duraciones largas y embalajes baratos, tienen mayor probabilidad de falla.

Esto permitir√° que, cuando tu socio haga la demo, el modelo realmente "prediga" cosas coherentes.

### üíª generate_demo_data.py

Python

```
import pandas as pd
import numpy as np
import random

def generate_logistics_data(n_samples=1000):
    # Opciones para las variables
    cities = ['Panama City', 'Miami', 'Bogota', 'Madrid', 'Mexico City']
    packaging = ['TermoBox-A1', 'TermoBox-A2', 'Standard-C1', 'EcoCool-X']
    carriers = ['DHL', 'FedEx', 'Copa Cargo', 'Lufthansa']
    
    data = []
    
    for i in range(n_samples):
        # Generaci√≥n de variables independientes
        id_embarque = f"SHIP-{10000 + i}"
        origen = random.choice(cities)
        destino = random.choice([c for c in cities if c != origen])
        pkg = random.choice(packaging)
        carrier = random.choice(carriers)
        duration = round(random.uniform(5, 48), 1) # Entre 5 y 48 horas
        mes = random.randint(1, 12)
        valor = random.randint(5000, 100000)
        
        # --- L√ìGICA PARA EL "RIESGO" (Para que el modelo aprenda algo real) ---
        # Aumentamos riesgo si: dura mucho, es verano (jun-ago) o el empaque es "Standard-C1"
        risk_score = (duration / 48) * 0.4
        if mes in [6, 7, 8]: risk_score += 0.3
        if pkg == 'Standard-C1': risk_score += 0.3
        
        # Variable Objetivo: 1 (Excursi√≥n) o 0 (OK)
        excursion = 1 if risk_score + random.uniform(-0.1, 0.1) > 0.65 else 0
        
        data.append([id_embarque, origen, destino, pkg, carrier, duration, mes, valor, excursion])

    # Crear DataFrame
    columns = ['ID_Embarque', 'Origen_Ciudad', 'Destino_Ciudad', 'Tipo_Embalaje', 
               'Aerolinea_Carrier', 'Duraci√≥n_Est_Hrs', 'Mes_Embarque', 'Valor_Mercanc√≠a', 'Temp_Excursion']
    
    df = pd.DataFrame(data, columns=columns)
    
    # Guardar para el socio
    df.to_csv("datos_demo_logistica.csv", index=False)
    print(f"‚úÖ Archivo 'datos_demo_logistica.csv' generado con {n_samples} registros.")

if __name__ == "__main__":
    generate_logistics_data()
```
